SameNumbersFromTwoArrays.java
package arrays;

import java.util.Arrays;

public class SameNumbersFromTwoArrays {
    public static void main(String[] args) {
        System.out.println("===============================");
        SameNumbersFromTwoArrays.byMethod();
        System.out.println("===============================");
        SameNumbersFromTwoArrays.byConditionStatement();
        System.out.println("===============================");
        SameNumbersFromTwoArrays.byForLoop();
        System.out.println("===============================");
    }
    
    public static void byMethod() {
        int a[] = {1, 2, 3, 4, 5, 6};
        int b[] = {1, 2, 3, 4, 5, 6};
        
        boolean status = Arrays.equals(a, b);
        
        if (status == true) {
            System.out.println("Equal Arrays");
        } else {
            System.out.println("Not Equal Arrays");
        }
    }
    
    public static void byConditionStatement() {
        int a[] = {1, 2, 3, 4, 5, 6, 7};
        int b[] = {1, 2, 3, 4, 7, 6};
        
        if (a.length == b.length) {
            for (int i = 0; i < a.length; i++) {
                if (a[i] == b[i]) {
                    System.out.print(a[i]);
                }
            }
        } else {
            System.out.println("Not equal Arrays");
        }
    }
    
    public static void byForLoop() {
        int a[] = {5, 6, 6, 4, 7, 9, 10};
        int b[] = {1, 2, 3, 4, 7, 6};
        
        for (int i = 0; i < a.length; i++) {
            for (int j = 0; j < b.length; j++) {
                if (a[i] == b[j]) {
                    System.out.print(a[i] + " ");
                }
            }
        }
        System.out.println();
    }
}

package demo;
import java.util.HashSet;
import java.util.Set;

public class Interview {
    public static void main(String[] args) {
        int array[] = {1, 3, 1, 1, 2, 1, 2, 4};
        
        Set<Integer> set = new HashSet<>();
        
        for (int arrayValues : array) { 
            set.add(arrayValues);
        }
        
        for (int setvalue : set) {
            int count = 0;
            for (int arrayValues : array) {
                if (setvalue == arrayValues) {
                    count++;
                }
            }
            System.out.println("Count of " + setvalue + " is: " + count);
        }
    }    
}

package arrays;

public class AdditionOfArrays {
    public static void main(String[] args) {
        System.out.println("===================");
        AdditionOfArrays.byForLoop();
        System.out.println("===================");
        AdditionOfArrays.byWhileLoop();
        System.out.println("===================");
    }
    
    public static void byForLoop() {
        int a[] = {3, 7, 10, 20};
        int b[] = {8, 9, 6, 5};
        int sum;
        
        for (int i = 0; i < a.length; i++) {
            for (int j = 0; j < b.length; j++) {
                sum = a[i] + b[j];
                System.out.print(sum + "  ");
                i++;
            }
        }
        
        System.out.println();    
    }
    
    public static void byWhileLoop() {
        int a[] = {3, 7, 10, 20};
        int b[] = {8, 9, 6, 5};
        int sum;
        int n = 0;
        
        while (n < a.length) {
            sum = a[n] + b[n];
            System.out.print(sum + "  ");
            n++;    
        }    
        System.out.println();
    }
}

AscendingDescendingOrder.java
package arrays;

public class AscendingDescendingOrder {
    public static void main(String[] args) {
        System.out.println("==========================");
        AscendingDescendingOrder.withoutUsingVariable();
        System.out.println("==========================");
        AscendingDescendingOrder.withUsingVariable();
        System.out.println("==========================");   
    }
    
    public static void withoutUsingVariable() {
        int a[] = {200, 100, 400, 500, 150};
        
        for (int i = 0; i < a.length; i++) {
            for (int j = i + 1; j < a.length; j++) {
                if (a[i] >= a[j]) {
                    a[i] = a[i] + a[j];
                    a[j] = a[i] - a[j];
                    a[i] = a[i] - a[j]; 
                }
            }
        }
        
        for (int k = 0; k < a.length; k++) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
        
        for (int k = a.length - 1; k >= 0; k--) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
    }
    
    public static void withUsingVariable() {
        int a[] = {200, 100, 400, 500, 150};
        int x;
        
        for (int i = 0; i < a.length; i++) {
            for (int j = i + 1; j < a.length; j++) {
                if (a[i] >= a[j]) {
                    x = a[i];
                    a[i] = a[j];
                    a[j] = x;
                }
            }
        }
        
        for (int k = 0; k < a.length; k++) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
        
        for (int k = a.length - 1; k >= 0; k--) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
    }   
}

Swap Elements in 2D Array
Swap2DArray.java
public class Swap2DArray {
    public static void main(String[] args) {
        int[][] array = {
            {1, 2},
            {4, 5},
            {7, 8}
        };
        
        for (int arrayIndex = 0; arrayIndex < array.length; arrayIndex++) {
            int startIndex = 0; 
            int endIndex = 1;
            
            //swap numbers in 2D array
            int x = array[arrayIndex][startIndex];
            array[arrayIndex][startIndex] = array[arrayIndex][endIndex];
            array[arrayIndex][endIndex] = x;        
        }

        //Print 2D after swapping
        for (int i = 0; i < array.length; i++) {
            for (int j = 0; j < array[i].length; j++) {
                System.out.print(array[i][j] + " ");
            }
            System.out.println();
        }
    }
}
Output
2 1 5 4 8 7
