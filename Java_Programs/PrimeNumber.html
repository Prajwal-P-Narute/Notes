Java Program with Interactive Demo

PrimeNumbers.java
package numbers;

import java.util.Scanner;

public class PrimeNumbers {

    public static void main(String[] args) {
        System.out.println("================");
        PrimeNumbers.byForLoop();
        System.out.println("=================");
        PrimeNumbers.byWhileLoop();
        System.out.println("=================");    
    }
    
    public static void byForLoop() {
        int c = 0;
        for (int i = 1; i <= 50; i++) {
            for (int j = 2; j < i; j++) {
                if (i % j == 0) {
                    c = 1;
                }  
            }
            
            if (c != 1) {
                System.out.println(i + "  This is a Prime Number");
            } else {
                c = 0;
            }
        }    
    }
    
    public static void byWhileLoop() {
        Scanner scan = new Scanner(System.in);
        System.out.println("Enter Number to check weather is prime number or not");
        
        int a = scan.nextInt();
        int n = a - 1;
        int c = 0;
        while (n > 1) {
            if (a % n == 0) {
                c = 1;
            }
            n--;
        }
        if (c == 0) {
            System.out.println(a + " This is Prime Number");
        } else {
            System.out.println(a + "  This is a not Prime Number");
        }    
    }
}
