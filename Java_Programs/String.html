How it works

This tool analyzes your input text and counts:

  • Alphabets: Letters from A to Z (both uppercase and lowercase)
  • Numbers: Digits from 0 to 9
  • Whitespace: Spaces, tabs, and line breaks
  • Other: Any other characters (punctuation, symbols, etc.)
// Java code to count characters in a string public class CharacterCounter { public static void main(String[] args) { String input = "Hello World 123!"; int alphabets = 0; int numbers = 0; int whitespace = 0; int other = 0; for (int i = 0; i < input.length(); i++) { char c = input.charAt(i); if (Character.isLetter(c)) { alphabets++; } else if (Character.isDigit(c)) { numbers++; } else if (Character.isWhitespace(c)) { whitespace++; } else { other++; } } System.out.println("Alphabets: " + alphabets); System.out.println("Numbers: " + numbers); System.out.println("Whitespace: " + whitespace); System.out.println("Other: " + other); } }

package string;

public class ReverseString {

    public static void main(String[] args) {
        String a = "Pratik";
        for (int i = a.length()-1; i>=0; i--) {
            char b = a.charAt(i);
            System.out.print(b + " ");
        }
    }
}

// Second approach to reverse string by StringBuilder
package string;

public class ReverseString {

    public static void main(String[] args) {
        String a = "Pratik";
        StringBuilder sb = new StringBuilder();
        sb.append(a);
        sb.reverse();
        System.out.println(sb);
    }
}

import java.util.HashMap;

public class DuplicateWords {
    public static void main(String[] args) {
        String str = "Hello world hello hello";
        
        // Split the string into words
        String[] strArray = str.split(" ");
        
        // Create a HashMap to store the unique element as a key and value pair
        HashMap<String, Integer> map = new HashMap<>();

        // Add element of string into map
        for (String word : strArray) {
            map.put(word, null);
        }
        
        for (String mapKey : map.keySet()) {
            int count = 0;
            for (int a = 0; a < strArray.length; a++) {
                if ((strArray[a]).equals(mapKey)) {
                    count++;
                }
            }
            System.out.println("Count of key '" + mapKey + "' is: " + count);
        }
    }
}

import java.util.HashMap;

public class DuplicateChars {
    public static void main(String[] args) {
        String str = "Pratik Narute";
        
        // Create a HashMap to store the unique element as a key and value pair
        HashMap<Character, Integer> map = new HashMap<>();

        // Add element of string into map
        for (int a = 0; a < str.length(); a++) {
            map.put(str.charAt(a), null);
        }
        
        for (char mapKey : map.keySet()) {
            int count = 0;
            for (int a = 0; a < str.length(); a++) {
                if (str.charAt(a) == mapKey) {
                    count++;
                }
            }
            System.out.println("Count of Char '" + mapKey + "' is: " + count);
        }
    }
}
