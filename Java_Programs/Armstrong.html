ArmstrongNumber.java
package numbers;

import java.io.InputStream;
import java.util.Scanner;

public class ArmstrongNumber {

    public static void main(String[] args) {
        System.out.println("======================");
        ArmstrongNumber.multipleNumber();
        System.out.println("======================");
        ArmstrongNumber.singleNumber();    
        System.out.println("======================");
    }
    
    public static void singleNumber() {
        Scanner scan = new Scanner(System.in);
        System.out.println("Enter Number to check weather is Armstrong Number or Not");
        
        int a = scan.nextInt();
        int n = a, count = 0, rem;
        double sum = 0;
        
        while (n > 0) {
            count++;
            n = n / 10;
        }
        n = a;
        while (n > 0) {
            rem = n % 10;
            sum = sum + Math.pow(rem, count);
            n = n / 10;
        }
        
        if (a == sum) {
            System.out.println(a + " this is an Armstrong Number");
        } else {
            System.out.println(a + " this is not an Armstrong Number");
        }    
    }
    
    public static void multipleNumber() {
        for (int i = 1; i <= 1000; i++) {
            int n = i, count = 0, rem = 0;
            double sum = 0;
            
            while (n > 0) {
                count++;
                n = n / 10;
            }
            n = i;
            while (n > 0) {
                rem = n % 10;
                sum = sum + Math.pow(rem, count);
                n = n / 10;    
            }
            if (i == sum) {
                System.out.println(i + " it is an Armstrong Number");
            }
        }
    }
}

What is an Armstrong number?

An Armstrong number is a number that is equal to the sum of its own digits each raised to the power of the number of digits.

Example: 153 = 1³ + 5³ + 3³ = 1 + 125 + 27 = 153
