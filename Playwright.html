Navigate to Project Folder

mkdir WEB_PLAYWRIGHT_JS

Initialize npm

ðŸ”¹ 2. Install Playwright

npm install -D @playwright/test

Step 4: Create Utility.js

Create Base/Utility.js with MongoDB connection functions:

// Base/Utility.js
const { MongoClient } = require('mongodb');
require('dotenv').config();

const uri = process.env.MONGO_URI;
const dbName = process.env.DB_NAME;
let client;

async function connectToMongoDB() {
  console.log(`Connecting MongoDB:\nURL: ${uri}\nDBName: ${dbName}`);
  if (!client) {
    client = new MongoClient(uri);
    await client.connect();
    console.log('â˜‘ Connected to MongoDB at ${uri}');
    console.log('-'.repeat(100));
  }
  return client.db(dbName);
}

async function closeMongoDB() {
  if (client) {
    await client.close();
    console.log('â˜‘ Closed MongoDB connection');
    client = null;
    console.log('-'.repeat(100));
  }
}

async function getCollectionDataFromMongo(collectionName, keyName, keyValue) {
  try {
    const db = await connectToMongoDB();
    const query = { [keyName]: keyValue }; // âœ“ dynamic key
    const data = await db.collection(collectionName).find(query).toArray();
    if (data.length === 0) {
      console.log(`No data found for: ${keyName} = ${keyValue}`);
      console.log('-'.repeat(100));
      return [];
    }
    console.log(`Retrieved ${data.length} records for ${keyName} = ${keyValue}`);
    console.log(`\nFull ${collectionName} JSON:\n${JSON.stringify(data, null, 2)}`);
    console.log('-'.repeat(100));
    return data;
  } catch (error) {
    console.error(`âœ— Error fetching data from '${collectionName}': ${error.message}`);
    console.log('-'.repeat(100));
    return null;
  } finally {
    await closeMongoDB();
  }
}

async function validateDataPresence(data, key, expectedValue) {
  process.stdout.write(`Verifying: ${expectedValue} present in ${key}...\n`);
  const found = data.some(item => item[key] === expectedValue);
  if (found) {
    console.log('âœ“ Validation successful');
  } else {
    console.log('âœ— Validation failed');
  }
  console.log('-'.repeat(100));
  return found;
}

module.exports = {
  getCollectionDataFromMongo,
  validateDataPresence
};

Example of using these utilities in a test:

  // Your Playwright test code here

});
