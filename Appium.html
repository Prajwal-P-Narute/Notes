4. Java Project Setup

Tools: JDK 17, Maven, IntelliJ/Eclipse

7. Key Components

‚úÖ BaseSetup (Driver Initialization)

// BaseSetup.java
public class BaseSetup {
  public static AppiumDriver driver;

  public static void initDriver(String platform) throws MalformedURLException {
    if (platform.equalsIgnoreCase("android")) {
      UiAutomator2Options options = new UiAutomator2Options();
      options.setDeviceName("emulator-5554");
      options.setAppPackage("com.example.app");
      options.setAppActivity("com.example.app.MainActivity");
      driver = new AndroidDriver(new URL("http://127.0.0.1:4723"), options);
    } else if (platform.equalsIgnoreCase("ios")) {
      XCUITestOptions options = new XCUITestOptions();
      options.setDeviceName("iPhone 14");
      options.setBundleId("com.example.iosapp");
      driver = new IOSDriver(new URL("http://127.0.0.1:4723"), options);
    }
  }
}

‚úÖ Page Object Example

// LoginPage.java
public class LoginPage {
  private AppiumDriver driver;

  By username = By.id("com.example.app:id/username");
  By password = By.id("com.example.app:id/password");
  By loginBtn = By.id("com.example.app:id/login");

  public LoginPage(AppiumDriver driver) {
    this.driver = driver;
  }

  public void login(String user, String pass) {
    driver.findElement(username).sendKeys(user);
    driver.findElement(password).sendKeys(pass);
    driver.findElement(loginBtn).click();
  }
}

‚úÖ Step Definition

// LoginSteps.java
public class LoginSteps {
  LoginPage loginPage = new LoginPage(BaseSetup.driver);

  @Given("I launch the app")
  public void launchApp() {
    // App launch handled in hooks
  }

  @When("I login with {string} and {string}")
  public void iLogin(String user, String pass) {
    loginPage.login(user, pass);
  }

  @Then("I should see the home page")
  public void verifyHomePage() {
    Assert.assertTrue(BaseSetup.driver.findElement(By.id("com.example.app:id/home")).isDisplayed());
  }
}

‚úÖ Hooks (Before/After Test)

// Hooks.java
public class Hooks {
  @Before
  public void setUp() throws MalformedURLException {
    BaseSetup.initDriver("android"); // or ios
  }

  @After
  public void tearDown() {
    if (BaseSetup.driver != null) {
      BaseSetup.driver.quit();
    }
  }
}
