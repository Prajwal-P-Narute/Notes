Armstrong Numbers

Check if a number is an Armstrong number

ArmstrongNumber.java
package numbers; import java.util.Scanner; public class ArmstrongNumber { public static void main(String[] args) { System.out.println("======================"); multipleNumber(); System.out.println("======================"); singleNumber(); System.out.println("======================"); } public static void singleNumber() { Scanner scan = new Scanner(System.in); System.out.println("Enter Number to check weather is Armstrong Number or Not"); int a = scan.nextInt(); int n = a, count = 0, rem; double sum = 0; while (n > 0) { count++; n = n / 10; } n = a; while (n > 0) { rem = n % 10; sum = sum + Math.pow(rem, count); n = n / 10; } if (a == sum) { System.out.println(a + " this is an Armstrong Number"); } else { System.out.println(a + " this is not an Armstrong Number"); } } // Additional method would follow... }

Palindrome Number

Determine if a number is a palindrome

PalindromeNumber.java
package numbers; public class PalindromeNumber { public static void main(String[] args) { int num = 121; int reversed = 0, remainder, original = num; while (num != 0) { remainder = num % 10; reversed = reversed * 10 + remainder; num /= 10; } if (original == reversed) { System.out.println(original + " is a palindrome."); } else { System.out.println(original + " is not a palindrome."); } } }
121 is a palindrome.

Find and check prime numbers

PrimeNumbers.java
package numbers; public class PrimeNumbers { public static void main(String[] args) { System.out.println("Prime numbers between 1 and 50:"); for (int i = 2; i <= 50; i++) { if (isPrime(i)) { System.out.print(i + " "); } } } public static boolean isPrime(int num) { if (num <= 1) { return false; } for (int i = 2; i <= Math.sqrt(num); i++) { if (num % i == 0) { return false; } } return true; } }

Sorting Arrays in Ascending and Descending Order

AscendingDescendingOrder.java
package arrays; public class AscendingDescendingOrder { public static void main(String[] args) { System.out.println("=========================="); withoutUsingVariable(); System.out.println("=========================="); withUsingVariable(); System.out.println("=========================="); } public static void withoutUsingVariable() { int a[] = {200,100,400,500,150}; for (int i = 0; i < a.length; i++) { for (int j = i + 1; j < a.length; j++) { if (a[i] >= a[j]) { a[i] = a[i] + a[j]; a[j] = a[i] - a[j]; a[i] = a[i] - a[j]; } } } for (int k = 0; k < a.length; k++) { System.out.print(a[k] + " "); } System.out.println(); for (int k = a.length - 1; k >= 0; k--) { System.out.print(a[k] + " "); } System.out.println(); } public static void withUsingVariable() { int a[] = {200,100,400,500,150}; int x; for (int i = 0; i < a.length; i++) { for (int j = i + 1; j < a.length; j++) { if (a[i] >= a[j]) { x = a[i]; a[i] = a[j]; a[j] = x; } } } for (int k = 0; k < a.length; k++) { System.out.print(a[k] + " "); } System.out.println(); for (int k = a.length - 1; k >= 0; k--) { System.out.print(a[k] + " "); } System.out.println(); } }

public class Swap2DArray { public static void main(String[] args) { int[][] array = { {1, 2}, {4, 5}, {7, 8} }; for (int arrayIndex = 0; arrayIndex < array.length; arrayIndex++) { int startIndex = 0; int endIndex = 1; // Swap numbers in 2D array int x = array[arrayIndex][startIndex]; array[arrayIndex][startIndex] = array[arrayIndex][endIndex]; array[arrayIndex][endIndex] = x; } // Print 2D after swapping for (int i = 0; i < array.length; i++) { for (int j = 0; j < array[i].length; j++) { System.out.print(array[i][j] + " "); } System.out.println(); } } }
