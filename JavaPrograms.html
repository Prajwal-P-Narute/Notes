    public static void main(String[] args) {
        System.out.println("======================");
        ArmstrongNumber.multipleNumber();
        System.out.println("======================");
        ArmstrongNumber.singleNumber();	
        System.out.println("======================");
    }
    
    public static void singleNumber() {
        Scanner scan = new Scanner(System.in);
        System.out.println("Enter Number to check weather is Armstrong Number or Not");
        
        int a = scan.nextInt();
        int n = a, count = 0, rem;
        double sum = 0;
        
        while (n > 0) {
            count++;
            n = n / 10;
        }
        n = a;
        while (n > 0) {
            rem = n % 10;
            sum = sum + Math.pow(rem, count);
            n = n / 10;
        }
        
        if (a == sum) {
            System.out.println(a + " this is an Armstrong Number");
        } else {
            System.out.println(a + " this is not an Armstrong Number");
        }	
    }
    
    public static void multipleNumber() {
        for (int i = 1; i <= 1000; i++) {
            int n = i, count = 0, rem = 0;
            double sum = 0;
            
            while (n > 0) {
                count++;
                n = n / 10;
            }
            n = i;
            while (n > 0) {
                rem = n % 10;
                sum = sum + Math.pow(rem, count);
                n = n / 10;	
            }
            if (i == sum) {
                System.out.println(i + " it is an Armstrong Number");
            }
        }
    }
}

        System.out.println("==========================");
        AscendingDescendingOrder.withUsingVariable();
        System.out.println("==========================");   
    }
    
    public static void withoutUsingVariable() {
        int a[] = {200,100,400,500,150};
        
        for (int i = 0; i < a.length; i++) {
            for (int j = i+1; j < a.length; j++) {
                if (a[i] >= a[j]) {
                    a[i] = a[i] + a[j];
                    a[j] = a[i] - a[j];
                    a[i] = a[i] - a[j]; 
                }
            }
        }
        
        for (int k = 0; k < a.length; k++) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
        for (int k = a.length-1; k >= 0; k--) {
            System.out.print(a[k] + " ");
        }
        System.out.println();
    }
    
    public static void withUsingVariable() {
        int a[] = {200,100,400,500,150};

public class Swap2DArray {
    public static void main(String[] args) {
        int[][] array = {
            {1, 2},
            {4, 5},
            {7, 8}
        };
        
        for (int arrayIndex = 0; arrayIndex < array.length; arrayIndex++) {
            int startIndex = 0; 
            int endIndex = 1;
            
            //swap numbers in 2D array
            int x = array[arrayIndex][startIndex];
            array[arrayIndex][startIndex] = array[arrayIndex][endIndex];
            array[arrayIndex][endIndex] = x;		
        }

        //Print 2D after swapping

public class StarPatterns {
    public static void main(String[] args) {
        int size = 5;
        
        rightTriangle(size);
        invertedRightTriangle(size);
        pyramid(size);
        diamond(size);
        hollowSquare(size);
        rightPascal(size);
        sandglass(size);
    }
    
    public static void rightTriangle(int n) {
        System.out.println("Right-Angled Triangle (Size: " + n + "):");
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                System.out.print("* ");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void invertedRightTriangle(int n) {
        System.out.println("Inverted Right-Angled Triangle (Size: " + n + "):");
        for (int i = n; i >= 1; i--) {
            for (int j = 1; j <= i; j++) {
                System.out.print("* ");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void pyramid(int n) {
        System.out.println("Pyramid Pattern (Size: " + n + "):");
        for (int i = 1; i <= n; i++) {
            // Print spaces
            for (int j = i; j < n; j++) {
                System.out.print(" ");
            }
            // Print stars
            for (int k = 1; k <= (2*i-1); k++) {
                System.out.print("*");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void diamond(int n) {
        System.out.println("Diamond Pattern (Size: " + n + "):");
        // Upper pyramid
        for (int i = 1; i <= n; i++) {
            for (int j = i; j < n; j++) {
                System.out.print(" ");
            }
            for (int k = 1; k <= (2*i-1); k++) {
                System.out.print("*");
            }
            System.out.println();
        }
        // Lower pyramid
        for (int i = n-1; i >= 1; i--) {
            for (int j = n; j > i; j--) {
                System.out.print(" ");
            }
            for (int k = 1; k <= (2*i-1); k++) {
                System.out.print("*");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void hollowSquare(int n) {
        System.out.println("Hollow Square Pattern (Size: " + n + "):");
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (i == 1 || i == n || j == 1 || j == n)
                    System.out.print("* ");
                else
                    System.out.print("  ");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void rightPascal(int n) {
        System.out.println("Right Pascal's Triangle (Size: " + n + "):");
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                System.out.print("* ");
            }
            System.out.println();
        }
        for (int i = n-1; i >= 1; i--) {
            for (int j = 1; j <= i; j++) {
                System.out.print("* ");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void sandglass(int n) {
        System.out.println("Sandglass Pattern (Size: " + n + "):");
        for (int i = n; i >= 1; i--) {
            for (int j = i; j < n; j++) {
                System.out.print(" ");
            }
            for (int k = 1; k <= (2*i-1); k++) {
                System.out.print("*");
            }
            System.out.println();
        }
        for (int i = 2; i <= n; i++) {
            for (int j = i; j < n; j++) {
                System.out.print(" ");
            }
            for (int k = 1; k <= (2*i-1); k++) {
                System.out.print("*");
            }
            System.out.println();
        }
        System.out.println();
    }
